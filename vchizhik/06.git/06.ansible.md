## Ansible inventory

```
---
allhost:
 children:
    linux_hosts:
      vars:
        ansible_user: ansible
      hosts:
        ubuntu:
          ansible_host: 192.168.56.101
        centos:
          ansible_host: 192.168.56.103
```

## Ansible Playbook 1

```
---
- hosts: allhost
  become: yes
  tasks:
    - name: System info
      vars:
       msg: |
         - "{{ ansible_distribution }}" | "{{ ansible_distribution_version }}"
         - "{{ ansible_mounts }}"
         - "RAM: total:{{ ansible_memory_mb.real.total }}" | "free: {{ ansible_memory_mb.real.free }}"
      debug:
        msg: "{{ msg.split('\n') }}"
```

## Ansible Playbook 2

```
---
- hosts: allhost
  become: yes
  tasks:
     - name: adduser
       user:
            name: "ansible"
     - name: copy-ssh
       authorized_key:
            user: "ansible"
            key: "{{ lookup('file','/home/ansible/.ssh/id_rsa.pub') }}"
     - name: NOPASSWD for USER
       lineinfile:
            path: /etc/sudoers
            line: ansible ALL=NOPASSWD:ALL
```
