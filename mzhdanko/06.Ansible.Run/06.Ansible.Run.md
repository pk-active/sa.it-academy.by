## Inventory file

```bash
vim inv.yaml

all_hosts:
  children:
    linux_hosts:
      vars:
        ansible_user: vagrant
      hosts:
        cent_01:
          ansible_host: 192.168.33.10
webservers:
  hosts:
    cent_01:

ansible-inventory -i inv.yaml --graph                                                                                                          
@all:
  |--@all_hosts:
  |  |--@linux_hosts:
  |  |  |--cent_01
  |--@ungrouped:
  |--@webservers:
  |  |--cent_01
```

## Create playbook to print out remote host parameters

```bash
vim first.yaml
---
- hosts: webservers
  tasks:
  - name: Print OS/version
    debug:
      msg: "{{ ansible_distribution }} | {{ ansible_distribution_version }}"
    tags:
      - os-version
  - name: Print mount point/capacity/used
    debug:
      msg: "{{ ansible_mounts }}"
    tags:
      - mount
  - name: Print RAM/capacity/free
    debug:
      msg: "{{ ansible_memory_mb }}"
    tags:
      - memory
```

## Create playbook for the ansible user

```bash
vim second.yaml

---
- hosts: webservers
  become: yes
  tasks:
  - name: Create user mzhdanko
    user:
      name: "mzhdanko"
      state: present
  - name: Copy ssh-key
    authorized_key:
      user: "mzhdanko"
      key: "{{ lookup('file', lookup('env','HOME') + '/.ssh/id_rsa.pub') }}"
  - name: Add sudo, NOPASSWD for user
    copy:
      content: "mzhdanko ALL=(ALL:ALL) NOPASSWD:ALL"
      dest: "/etc/sudoers.d/mzhdanko"
```	  
	  
Running playbook for the ansible user (second.yaml)
```bash
ansible-playbook second.yaml -i inv.yaml
```
